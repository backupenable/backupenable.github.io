        r=i+1

import sys
from math import comb

def solve():
    N, mo = map(int, sys.stdin.readline().split())
    a = list(map(int, sys.stdin.readline().split()))
    
    if N == 1:
        print(a[0] % mo)
        return
    
    total = 0
    
    def dfs(current_a):
        nonlocal total
        if len(current_a) == 1:
            total = (total + current_a[0]) % mo
            return
        for i in range(len(current_a)):
            for j in range(i + 1, len(current_a)):
                new_a = current_a.copy()
                y = new_a.pop(j)
                x = new_a.pop(i)
       
                new_a1 = new_a.copy()
                new_a1.append(x + y)
                dfs(new_a1)
        
                new_a2 = new_a.copy()
                new_a2.append(x * y)
                dfs(new_a2)
    
    dfs(a)
    
    print(total)
