def solve():

    a = list(map(int, sys.stdin.readline().split()))

    def dfs(current_a):
        nonlocal total
        if len(current_a) == 1:
            total = (total + current_a[0]) % mo
            return
        for i in range(len(current_a)):
            for j in range(i + 1, len(current_a)):
                new_a = current_a.copy()
                y = new_a.pop(j)
                x = new_a.pop(i)
       
                new_a1 = new_a.copy()
                new_a1.append(x + y)
                dfs(new_a1)
        
                new_a2 = new_a.copy()
                new_a2.append(x * y)
                dfs(new_a2)
    
    dfs(a)
