def knapsack(weights, values, capacity):
    n = len(weights)
    # ÂàõÂª∫‰∫åÁª¥Êï∞ÁªÑdp
    dp = [[0 for _ in range(capacity + 1)] for _ in range(n + 1)]
    
    # Â°´ÂÖÖdpÊï∞ÁªÑ
    for i in range(1, n + 1):
        for w in range(1, capacity + 1):
            if weights[i-1] <= w:
                # ÈÄâÊã©ÂΩìÂâçÁâ©ÂìÅÊàñ‰∏çÈÄâÊã©ÔºåÂèñÊúÄÂ§ßÂÄº
                dp[i][w] = max(dp[i-1][w], dp[i-1][w-weights[i-1]] + values[i-1])

                # ÂΩìÂâçÁâ©ÂìÅÈáçÈáèË∂ÖËøáËÉåÂåÖÂÆπÈáèÔºå‰∏çËÉΩÈÄâÊã©
                dp[i][w] = dp[i-1][w]
    
    # ËøîÂõûÊúÄÂ§ß‰ª∑ÂÄº
    return dp[n][capacity]

def tsp(dist):
    n = len(dist)
    # dp[i][mask]Ë°®Á§∫‰ªéËµ∑ÁÇπ0Âá∫ÂèëÔºåÂΩìÂâçÂú®ÂüéÂ∏ÇiÔºåÂ∑≤ËÆøÈóÆÂüéÂ∏ÇÈõÜÂêà‰∏∫maskÊó∂ÁöÑÊúÄÁü≠Ë∑ØÂæÑÈïøÂ∫¶

    dp = [[float('inf')] * (1 << n) for _ in range(n)]
    
    # ÂàùÂßãÂåñÔºö‰ªéËµ∑ÁÇπ0Âá∫ÂèëÔºåÂ∑≤ËÆøÈóÆËµ∑ÁÇπ0Ôºàmask=1ÔºâÔºåË∑ØÂæÑÈïøÂ∫¶‰∏∫0
    dp[0][1] = 0
    
    # Êûö‰∏æÊâÄÊúâÂèØËÉΩÁöÑÂüéÂ∏ÇËÆøÈóÆÁä∂ÊÄÅ
    for mask in range(1 << n):
        # ÈÅçÂéÜÂΩìÂâçÂ∑≤ËÆøÈóÆÁöÑÂüéÂ∏Ç
        for current_city in range(n):
            # Â¶ÇÊûúÂΩìÂâçÂüéÂ∏Ç‰∏çÂú®Â∑≤ËÆøÈóÆÈõÜÂêà‰∏≠ÔºåË∑≥Ëøá

            # Â∞ùËØï‰ªéÊú™ËÆøÈóÆÁöÑÂüéÂ∏Ç‰∏≠ÈÄâÊã©‰∏ã‰∏Ä‰∏™Ë¶ÅËÆøÈóÆÁöÑÂüéÂ∏Ç
            for next_city in range(n):
                # Â¶ÇÊûú‰∏ã‰∏Ä‰∏™ÂüéÂ∏ÇÂ∑≤ÁªèËÆøÈóÆËøáÔºåË∑≥Ëøá

                new_mask = mask | (1 << next_city)
                # Êõ¥Êñ∞‰ªéÂΩìÂâçÂüéÂ∏ÇÂà∞‰∏ã‰∏Ä‰∏™ÂüéÂ∏ÇÁöÑÊúÄÁü≠Ë∑ØÂæÑ

                    dp[next_city][new_mask],
                    dp[current_city][mask] + dist[current_city][next_city]
                )
    
    # ÊâæÂà∞ËÆøÈóÆÂÆåÊâÄÊúâÂüéÂ∏ÇÂêéÔºå‰ªéÊúÄÂêé‰∏Ä‰∏™ÂüéÂ∏ÇËøîÂõûËµ∑ÁÇπÁöÑÊúÄÁü≠Ë∑ØÂæÑ
    ans = float('inf')
    for city in range(n):
        # Â¶ÇÊûúÂüéÂ∏Ç‰∏çÊòØËµ∑ÁÇπÔºàÂõ†‰∏∫Â∑≤ÁªèËÆøÈóÆ‰∫ÜÊâÄÊúâÂüéÂ∏ÇÔºåÊâÄ‰ª•maskÊòØÂÖ®1Ôºâ

def graph_coloring_dp(graph):
    """
    ‰ΩøÁî®Âä®ÊÄÅËßÑÂàíËß£ÂÜ≥ÂõæÁùÄËâ≤ÈóÆÈ¢ò
    ÂèÇÊï∞:
        graph: ÈÇªÊé•Ë°®Ë°®Á§∫ÁöÑÂõæÔºå‰æãÂ¶Ç[[1,2], [0,2], [0,1]]Ë°®Á§∫‰∏âËßíÂΩ¢Âõæ
    ËøîÂõû:
        ÊúÄÂ∞ëÈúÄË¶ÅÁöÑÈ¢úËâ≤Êï∞
    """
    n = len(graph)
    if n == 0:
        return 0
    
    # È¢ÑÂ§ÑÁêÜÊâÄÊúâÁã¨Á´ãÈõÜ(‰ΩøÁî®‰ΩçÊé©Á†ÅË°®Á§∫)
    independent_sets = []
    for mask in range(1, 1 << n):
        is_independent = True
        for i in range(n):
            if mask & (1 << i):
                for neighbor in graph[i]:
                    if mask & (1 << neighbor):
                        is_independent = False
                        break
                if not is_independent:
                    break
        if is_independent:
            independent_sets.append(mask)
    
    # ÂàùÂßãÂåñDPÊï∞ÁªÑ
    dp = [float('inf')] * (1 << n)
    dp[0] = 0  # Á©∫ÈõÜÈúÄË¶Å0ÁßçÈ¢úËâ≤
    
    # Âä®ÊÄÅËßÑÂàíÂ°´Ë°®
    for mask in range(1, 1 << n):
        # ÈÅçÂéÜÊâÄÊúâÁã¨Á´ãÂ≠êÈõÜ
        for subset in independent_sets:
            if (mask & subset) == subset:  # subsetÊòØmaskÁöÑÂ≠êÈõÜ
                dp[mask] = min(dp[mask], dp[mask ^ subset] + 1)
    
    return dp[(1 << n) - 1]

def set_cover(universe, subsets):
    m = len(universe)
    # Â∞ÜÂÖÉÁ¥†Êò†Â∞ÑÂà∞Á¥¢Âºï
    element_to_index = {element: i for i, element in enumerate(universe)}
    
    # Â∞ÜÊØè‰∏™Â≠êÈõÜËΩ¨Êç¢‰∏∫‰ΩçÊé©Á†Å
    subset_masks = []
    for subset in subsets:
        mask = 0
        for element in subset:
            if element in element_to_index:
                mask |= 1 << element_to_index[element]
        subset_masks.append(mask)
    
    # Âä®ÊÄÅËßÑÂàíÊ±ÇËß£
    dp = [float('inf')] * (1 << m)
    dp[0] = 0  # Á©∫ÈõÜ‰∏çÈúÄË¶Å‰ªª‰ΩïÂ≠êÈõÜ
    
    for mask in range(1 << m):
        if dp[mask] == float('inf'):
            continue
        for subset_mask in subset_masks:
            new_mask = mask | subset_mask
            dp[new_mask] = min(dp[new_mask], dp[mask] + 1)
    
    # ËøîÂõûË¶ÜÁõñÂÖ®ÈõÜÊâÄÈúÄÁöÑÊúÄÂ∞èÂ≠êÈõÜÊï∞
    return dp[(1 << m) - 1]

    {1, 2, 3},
    {2, 4},
    {3, 4},
    {4, 5}
]
print(set_cover(universe, subsets))  # ËæìÂá∫: 2

def hamiltonian_cycle(graph):
    n = len(graph)
    # ÂàùÂßãÂåñdpÊï∞ÁªÑ
    dp = [[False] * n for _ in range(1 << n)]
    
    # ÂàùÂßãÊù°‰ª∂ÔºöÂè™ËÆøÈóÆ‰∏Ä‰∏™È°∂ÁÇπ
    for i in range(n):
        dp[1 << i][i] = True
    
    # Áä∂ÊÄÅÂéãÁº©DP
    for mask in range(1 << n):
        for i in range(n):
            if not (mask & (1 << i)):  # i‰∏çÂú®mask‰∏≠
                continue
            # Êûö‰∏æmask‰∏≠Èô§iÂ§ñÁöÑÈ°∂ÁÇπj
            for j in range(n):
                if j == i:
                    continue
                if mask & (1 << j) and graph[j][i] and dp[mask ^ (1 << i)][j]:
                    dp[mask][i] = True
                    break
    
    # Ê£ÄÊü•ÊòØÂê¶Â≠òÂú®ÂìàÂØÜÂ∞îÈ°øÂõûË∑Ø
    for i in range(1, n):  # Ëµ∑ÁÇπ‰∏∫0ÔºåÊ£ÄÊü•ÂÖ∂‰ªñÈ°∂ÁÇπÂà∞0ÊòØÂê¶ÊúâËæπ
        if dp[(1 << n) - 1][i] and graph[i][0]:
            return True
    return False

    [0, 1, 1, 0]

def can_partition(nums):
    total_sum = sum(nums)
    # Â¶ÇÊûúÊÄªÂíå‰∏∫Â•áÊï∞ÔºåÊó†Ê≥ïÂπ≥ÂàÜ
    if total_sum % 2 != 0:
        return False
    
    target = total_sum // 2
    n = len(nums)
    
    # ÂàõÂª∫dpÊï∞ÁªÑ
    dp = [[False] * (target + 1) for _ in range(n + 1)]
    
    # ÂàùÂßãÊù°‰ª∂
    dp[0][0] = True  # Á©∫ÈõÜÁöÑÂíå‰∏∫0
    
    # Â°´ÂÖÖdpÊï∞ÁªÑ
    for i in range(1, n + 1):
        for s in range(target + 1):
            # ‰∏çÈÄâÊã©ÂΩìÂâçÂÖÉÁ¥†
            dp[i][s] = dp[i-1][s]
            # ÈÄâÊã©ÂΩìÂâçÂÖÉÁ¥†
            if s >= nums[i-1]:
                dp[i][s] = dp[i][s] or dp[i-1][s-nums[i-1]]
    
    return dp[n][target]

# Á§∫‰æã

print(can_partition(nums))  # ËæìÂá∫: True

def sat_solver(clauses, variables):
    """
    ‰ΩøÁî®Âä®ÊÄÅËßÑÂàíËß£ÂÜ≥SATÈóÆÈ¢ò
    
    ÂèÇÊï∞:
    clauses: ÂàóË°®ÔºåÊØè‰∏™ÂÖÉÁ¥†ÊòØ‰∏Ä‰∏™Â≠êÂè•ÔºåÂ≠êÂè•Ë°®Á§∫‰∏∫ÊñáÂ≠óÂàóË°®ÔºåÊ≠£ÊñáÂ≠ó‰∏∫Ê≠£Êï∞ÔºåË¥üÊñáÂ≠ó‰∏∫Ë¥üÊï∞
    variables: ÂèòÈáèÊï∞Èáè
    
    ËøîÂõû:
    ÊòØÂê¶Â≠òÂú®Êª°Ë∂≥ÊâÄÊúâÂ≠êÂè•ÁöÑÂèòÈáèËµãÂÄº
    """
    # È¢ÑÂ§ÑÁêÜÔºöËÆ°ÁÆóÊØè‰∏™Â≠êÂè•ÁöÑÁä∂ÊÄÅÊé©Á†Å
    clause_masks = []
    for clause in clauses:
        mask = 0
        for literal in clause:
            var = abs(literal) - 1  # ÂèòÈáèÁ¥¢Âºï‰ªé0ÂºÄÂßã
            if literal > 0:
                mask |= 1 << var  # Ê≠£ÊñáÂ≠óÂú®ÂØπÂ∫î‰ΩçÁΩÆ‰∏∫1
            else:
                mask |= 1 << (var + variables)  # Ë¥üÊñáÂ≠óÂú®È´òÂçäÈÉ®ÂàÜÂØπÂ∫î‰ΩçÁΩÆ‰∏∫1
        clause_masks.append(mask)
    
    # Áä∂ÊÄÅÊï∞ÈáèÔºöÊØè‰∏™ÂèòÈáèÂèØ‰ª•‰∏∫ÁúüÊàñÂÅá
    states = 1 << (2 * variables)
    
    # dp[i][s]Ë°®Á§∫ËÄÉËôëÂâçi‰∏™ÂèòÈáèÔºåÂ≠êÂè•Áä∂ÊÄÅ‰∏∫sÊó∂ÊòØÂê¶ÂèØÊª°Ë∂≥
    dp = [False] * states
    dp[0] = True  # ÂàùÂßãÁä∂ÊÄÅÔºöÊ≤°ÊúâÂèòÈáèÔºåÊâÄÊúâÂ≠êÂè•ÈÉΩÊú™Êª°Ë∂≥
    
    # ÈÅçÂéÜÊØè‰∏™ÂèòÈáè
    for var in range(variables):
        # Â§çÂà∂ÂΩìÂâçÁä∂ÊÄÅ
        new_dp = [False] * states
        for s in range(states):
            if not dp[s]:
                continue
            
            # Â∞ùËØïÂ∞ÜÂèòÈáèËÆæ‰∏∫Áúü
            true_mask = 1 << var
            new_state_true = s
            for i, mask in enumerate(clause_masks):
                if (mask & true_mask) and not (s & (1 << i)):
                    new_state_true |= 1 << i
            
            # Â∞ùËØïÂ∞ÜÂèòÈáèËÆæ‰∏∫ÂÅá
            false_mask = 1 << (var + variables)
            new_state_false = s
            for i, mask in enumerate(clause_masks):
                if (mask & false_mask) and not (s & (1 << i)):
                    new_state_false |= 1 << i
            
            new_dp[new_state_true] = True
            new_dp[new_state_false] = True
        
        dp = new_dp
    
    # Ê£ÄÊü•ÊòØÂê¶ÊâÄÊúâÂ≠êÂè•ÈÉΩË¢´Êª°Ë∂≥
    all_satisfied = (1 << len(clauses)) - 1
    return dp[all_satisfied]
