def set_cover(universe, subsets):
    m = len(universe)
    # å°†å…ƒç´ æ˜ å°„åˆ°ç´¢å¼•
    element_to_index = {element: i for i, element in enumerate(universe)}
    
    # å°†æ¯ä¸ªå­é›†è½¬æ¢ä¸ºä½æ©ç 
    subset_masks = []
    for subset in subsets:
        mask = 0
        for element in subset:
            if element in element_to_index:
                mask |= 1 << element_to_index[element]
        subset_masks.append(mask)
    
    # åŠ¨æ€è§„åˆ’æ±‚è§£
    dp = [float('inf')] * (1 << m)
    dp[0] = 0  # ç©ºé›†ä¸éœ€è¦ä»»ä½•å­é›†
    
    for mask in range(1 << m):

            continue

            new_mask = mask | subset_mask
            dp[new_mask] = min(dp[new_mask], dp[mask] + 1)
    
    # è¿”å›è¦†ç›–å…¨é›†æ‰€éœ€çš„æœ€å°å­é›†æ•°
    return dp[(1 << m) - 1]

def can_partition(nums):
    total_sum = sum(nums)
    # å¦‚æœæ€»å’Œä¸ºå¥‡æ•°ï¼Œæ— æ³•å¹³åˆ†
    if total_sum % 2 != 0:
        return False
    
    target = total_sum // 2
    n = len(nums)
    
    # åˆ›å»ºdpæ•°ç»„
    dp = [[False] * (target + 1) for _ in range(n + 1)]
    
    # åˆå§‹æ¡ä»¶
    dp[0][0] = True  # ç©ºé›†çš„å’Œä¸º0
    
    # å¡«å……dpæ•°ç»„
    for i in range(1, n + 1):
        for s in range(target + 1):
            # ä¸é€‰æ‹©å½“å‰å…ƒç´ 
            dp[i][s] = dp[i-1][s]
            # é€‰æ‹©å½“å‰å…ƒç´ 
            if s >= nums[i-1]:
                dp[i][s] = dp[i][s] or dp[i-1][s-nums[i-1]]
    
    return dp[n][target]
